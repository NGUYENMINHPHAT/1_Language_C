#include <stdio.h>
#include <math.h>
int main(){
  int c,b,a,s;
scanf("%d%d",&c,&b);
a=b-c;
if (a<0) printf("So nhap khong hop le");
else{
if (a<=50) s=a*1549;
if (a>50 && a<=100)  s=1549*50+(a-50)*1600;
if (a>100 && a<=200) s=1549*50+1600*50+(a-100)*1858;
if (a>200 && a<=300) s=1549*50+1600*50+100*1858+(a-200)*2340;
if (a>300 && a<=400) s=1549*50+1600*50+100*1858+2340*100+(a-300)*2615;
if (a>400)           s=1549*50+1600*50+100*1858+2340*100+2615*100+(a-400)*2701;
}
printf("%d Kwh: %d VND",a,s);
}
